\begin{exercicioBanco}[\valorquestao]
Uma máquina produz \(240\) peças em \(6\) horas, trabalhando sempre no mesmo ritmo. Quantas peças essa máquina produzirá em \(10\) horas?

% Define as alternativas
\newcommand{\alternativas}{%

        \begin{tabularx}{\textwidth}{XXXXX}
            (a) \(360\). &
            (b) \(380\). &
            (c) \(400\). &
            (d) \(420\). &
            (e) \(480\).
        \end{tabularx}
    \end{center}
}
