\begin{exercicioBanco}[\valorquestao]
Seja a função afim 
%
\[
    \begin{array}{rcl}
        f:\ \bbR & \to & \bbR \\
        x & \mapsto & -2x + 3.
    \end{array}
\]
%
Calcule o valor de \(\dfrac{1}{3}\cdot f(0) + 2\cdot f\left(\dfrac{3}{2}\right)\).

% Define as alternativas
\newcommand{\alternativas}{%
    \begin{center}
        \begin{tabularx}{\textwidth}{XXXXX}
            (a) \(1\). &
            (b) \(2\). &
            (c) \(3\). &
            (d) \(4\). &
            (e) \(5\).
        \end{tabularx}
    \end{center}
}

\end{exercicioBanco}
