\begin{exercicioBanco}[\valorquestao]
O valor a ser pago por um serviço de entrega é dado pela função \(f(x) = 5{,}00 + 1{,}80x\), sendo \(x\) a distância percorrida em km. Determine o valor a ser pago por uma entrega de 12 km.

% Define as alternativas
\newcommand{\alternativas}{%
    \begin{center}
        \begin{tabularx}{\textwidth}{XXXXX}
            (a) R\$ \(21{,}60\). &
            (b) R\$ \(25{,}00\). &
            (c) R\$ \(26{,}60\). &
            (d) R\$ \(27{,}60\). &
            (e) R\$ \(23{,}80\).
        \end{tabularx}
    \end{center}
}

\end{exercicioBanco}
