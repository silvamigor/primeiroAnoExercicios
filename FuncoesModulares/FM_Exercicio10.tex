\begin{exercicioBanco}[\valorquestao]
Determine o conjunto \(S\) dos números reais que satisfazem a equação \(|x| = x - 6\).

% Define as alternativas
\newcommand{\alternativas}{%
    \begin{center}
        \begin{tabularx}{\textwidth}{XXXXX}
            (a) \(S = [1, 2]\). &
            (b) \(S = [-2, -1]\). &
            (c) \(S = ]-1,1[\). &
            (d) \(S = \varnothing\). &
            (e) \(S = [0,\infty[\).
        \end{tabularx}
    \end{center}
}
