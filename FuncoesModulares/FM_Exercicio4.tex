\begin{exercicioBanco}[\valorquestao]
Considere a função \(f(x) = \dfrac{2}{|x - 3|}\). Calcule \(f\left(\dfrac{1}{2}\right)\).

% Define as alternativas

    \begin{center}
        \begin{tabularx}{\textwidth}{XXXXX}
            (a) \(-\dfrac{4}{5}\). &

            (c) \(\dfrac{10}{2}\). &
            (d) \(-\dfrac{10}{2}\). &
            (e) \(\dfrac{5}{2}\).
        \end{tabularx}
    \end{center}
}
